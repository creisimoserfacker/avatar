# basic tool making
country_event = {
	id = invention.1
	title = invention.1.t
	desc = invention.1.d
	picture = GFX_invention_event_001

	is_triggered_only = yes

	fire_only_once = yes
	minor_flavor = yes

	immediate = {
		hidden_effect = {
			set_technology = {
				basic_tool_making = 1
			}
			if = {
				limit = {
					NOT = {
						has_global_flag = basic_tool_making_event
					}
				}
				set_global_flag = basic_tool_making_event
				news_event = {
					id = invention_news.1
					days = 240
					random_days = 240
				}
			}
		}
	}

	option = {
		name = invention.1.a
		trigger = { 
			NOT = {
				any_country = {
					has_tech = basic_tool_making
				}
			}
		}

	}
	option = {
		name = invention.1.b
		trigger = { 
			any_country = {
				has_tech = basic_tool_making
			}
		}

	}

}
