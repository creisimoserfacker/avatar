peasant_mood_increase = {
	if = {
		limit = {
			has_idea = integrated_peasants
		}
	}
	else_if = {
		limit = {
			has_idea = happy_peasants
		}
		swap_ideas = {
			remove_idea = happy_peasants
			add_idea = integrated_peasants
		}
	}
	else_if = {
		limit = {
			has_idea = pleased_peasants
		}
		swap_ideas = {
			remove_idea = pleased_peasants
			add_idea = happy_peasants
		}
	}
	else_if = {
		limit = {
			has_idea = distrusting_peasants
		}
		remove_ideas = distrusting_peasants
	}
	else_if = {
		limit = {
			has_idea = angered_peasants
		}
		swap_ideas = {
			remove_idea = angered_peasants
			add_idea = distrusting_peasants
		}
	}
	else_if = {
		limit = {
			has_idea = enraged_peasants
		}
		swap_ideas = {
			remove_idea = enraged_peasants
			add_idea = angered_peasants
		}
	}
	else = {
		add_ideas = pleased_peasants
	}
}

peasant_opinion_recalc = {
	remove_ideas = integrated_peasants
	remove_ideas = happy_peasants
	remove_ideas = pleased_peasants
	remove_ideas = distrusting_peasants
	remove_ideas = angered_peasants
	remove_ideas = enraged_peasants
	if = {
		limit = {
			check_variable = { modifier@peasant_opinion > 24 }
		}
		add_ideas = integrated_peasants
	}
	else_if = {
		limit = {
			check_variable = { modifier@peasant_opinion > 14 }
		}
		add_ideas = happy_peasants
	}
	else_if = {
		limit = {
			check_variable = { modifier@peasant_opinion > 9 }
		}
		add_ideas = pleased_peasants
	}
	
	else_if = {
		limit = {
			check_variable = { modifier@peasant_opinion < -9 }
		}
		add_ideas = enraged_peasants
	}
	else_if = {
		limit = {
			check_variable = { modifier@peasant_opinion < -14 }
		}
		add_ideas = angered_peasants
	}
	else_if = {
		limit = {
			check_variable = { modifier@peasant_opinion < -24 }
		}
		add_ideas = distrusting_peasants
	}
	else = {
		
	}
}
